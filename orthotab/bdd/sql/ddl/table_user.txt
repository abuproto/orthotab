CREATE TABLE USER
(
    id       INT PRIMARY KEY AUTO_INCREMENT,
    login    VARCHAR(30),
    nom    VARCHAR(30),
    prenom    VARCHAR(30),
    password    VARCHAR(10),
    role         VARCHAR(30)
);
