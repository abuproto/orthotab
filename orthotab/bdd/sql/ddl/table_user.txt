CREATE TABLE USER
(
    id       INT PRIMARY KEY AUTO_INCREMENT,
    login    VARCHAR(30),
    nom    VARCHAR(30),
    prenom    VARCHAR(30),
    password    VARCHAR(10),
    role         VARCHAR(30),
    nivcourant	INT default 1
);


GRANT SELECT ON USER TO PUBLIC;
GRANT UPDATE ON USER TO PUBLIC;
GRANT DELETE ON USER TO PUBLIC;
GRANT INSERT ON USER TO PUBLIC;
